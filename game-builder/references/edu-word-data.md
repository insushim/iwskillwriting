// 학년별 로드
export function getWordsForGrade(grade: number): WordData[] {
  return WORD_DATA[grade] || WORD_DATA[3];
}

// 랜덤 퀴즈 생성
export function generateQuiz(words: WordData[], count: number = 4): {
  question: WordData;
  options: string[];
  correctIndex: number;
} {
  const shuffled = [...words].sort(() => Math.random() - 0.5);
  const question = shuffled[0];
  const wrongAnswers = shuffled.slice(1, count).map(w => w.ko);
  const options = [...wrongAnswers, question.ko].sort(() => Math.random() - 0.5);
  return {
    question,
    options,
    correctIndex: options.indexOf(question.ko),
  };
}
